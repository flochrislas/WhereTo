<script type="text/javascript">
  $('#tag-input').each(function() {
      var $this = $(this);
      var src = $this.data('action');

      $this.autocomplete({
          source: src,
          minLength: 2,
          select: function(event, ui) {
              $this.val(ui.item.value);
              $('#tag-selection').val(ui.item.id);
              $('#tag-search').val(ui.item.value);
          }
      });
  });
</script>
